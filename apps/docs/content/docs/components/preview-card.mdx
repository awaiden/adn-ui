```tsx
import { PreviewCard } from "@adn-ui/react";
```

### Default

<ComponentPreview name="preview-card" />

## Styling

### Customizing the component classes

```css
.preview-card__popup {
  @apply bg-popover text-popover-foreground data-[state=open]:animate-in data-[state=closed]:animate-out data-[state=closed]:fade-out-0 data-[state=open]:fade-in-0 data-[state=closed]:zoom-out-95 data-[state=open]:zoom-in-95 data-[side=bottom]:slide-in-from-top-2 data-[side=left]:slide-in-from-right-2 data-[side=right]:slide-in-from-left-2 data-[side=top]:slide-in-from-bottom-2 z-50 w-64 rounded-md border p-4 shadow-md outline-none;
}
```

## CSS Classes

### Slots

- `preview-card`: The class for the root slot.
- `preview-card__trigger`: The class for the trigger slot.
- `preview-card__portal`: The class for the portal slot.
- `preview-card__backdrop`: The class for the backdrop slot.
- `preview-card__positioner`: The class for the positioner slot.
- `preview-card__popup`: The class for the popup slot.
- `preview-card__arrow`: The class for the arrow slot.
