Renders an accessible label associated with controls.

## Preview

<ComponentPreview>
  <LabelExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { Label } from "@adn-ui/react";

export default () => (
  <div>
    <Label htmlFor="email">Email</Label>
    <Input id="email" />
  </div>
);
```
