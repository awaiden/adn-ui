A two-state button that can be either on or off.

## Preview

<ComponentPreview>
  <ToggleExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { Toggle } from "@adn-ui/react";
import { Bold } from "lucide-react";

export default () => (
  <Toggle aria-label="Toggle bold">
    <Bold size={16} />
  </Toggle>
);
```
