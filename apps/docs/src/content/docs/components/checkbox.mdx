<ComponentPreview>
  <CheckboxExample.WithLabel client:load />
</ComponentPreview>

## Usage

```tsx
import { Checkbox } from "@adn-ui/react";

export default () => (
  <Checkbox.Root>
    <Checkbox.Indicator />
  </Checkbox.Root>
);
```

Renders when the checkbox is in a checked or indeterminate state.
