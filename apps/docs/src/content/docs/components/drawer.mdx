A panel that slides in from the edge of the screen.

## Preview

<ComponentPreview>
  <DrawerExample.Right client:load />
</ComponentPreview>

## Usage

```tsx
import { Button, Drawer } from "@adn-ui/react";

export default () => (
  <Drawer.Root>
    <Drawer.Trigger asChild>
      <Button variant="outline">Open Drawer</Button>
    </Drawer.Trigger>
    <Drawer.Content>
      <Drawer.Header>
        <Drawer.Title>Title</Drawer.Title>
        <Drawer.Description>Description goes here.</Drawer.Description>
      </Drawer.Header>
    </Drawer.Content>
  </Drawer.Root>
);
```
