Displays a callout for important information.

## Preview

<ComponentPreview>
  <AlertExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { Alert } from "@adn-ui/react";
import { Terminal } from "lucide-react";

export default () => (
  <Alert.Root>
    <Alert.Icon>
      <Terminal />
    </Alert.Icon>
    <Alert.Title>Heads up!</Alert.Title>
    <Alert.Description>You can add components using the CLI.</Alert.Description>
  </Alert.Root>
);
```

### Destructive

Use the destructive variant to communicate errors or dangerous actions.

<ComponentPreview>
  <AlertExample.Destructive client:load />
</ComponentPreview>

```tsx
<Alert.Root variant="destructive">
  <Alert.Icon>
    <AlertCircle />
  </Alert.Icon>
  <Alert.Title>Error</Alert.Title>
  <Alert.Description>Your session has expired. Please log in again.</Alert.Description>
</Alert.Root>
```
