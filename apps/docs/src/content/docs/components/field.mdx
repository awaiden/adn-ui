A form field wrapper that groups a label, input, description, and error message.

## Preview

<ComponentPreview>
  <FieldExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { Field, Input } from "@adn-ui/react";

export default () => (
  <Field.Root id="email">
    <Field.Label>Email</Field.Label>
    <Input id="email" type="email" placeholder="you@example.com" />
  </Field.Root>
);
```
