<ComponentPreview>
  <TableExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { Table } from "@adn-ui/react";

export default () => (
  <Table.Root>
    <Table.Header>
      <Table.Row>
        <Table.Head>Name</Table.Head>
        <Table.Head>Email</Table.Head>
      </Table.Row>
    </Table.Header>
    <Table.Body>
      <Table.Row>
        <Table.Cell>Alice</Table.Cell>
        <Table.Cell>alice@example.com</Table.Cell>
      </Table.Row>
    </Table.Body>
  </Table.Root>
);
```
