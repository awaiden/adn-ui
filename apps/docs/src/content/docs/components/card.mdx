Displays a card with header, content, and footer.

<ComponentPreview>
  <CardExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { Card } from "@adn-ui/react";

export default () => (
  <Card.Root>
    <Card.Header>
      <Card.Title>Card Title</Card.Title>
      <Card.Description>Card Description</Card.Description>
    </Card.Header>
    <Card.Content>
      <p>Card Content</p>
    </Card.Content>
  </Card.Root>
);
```
