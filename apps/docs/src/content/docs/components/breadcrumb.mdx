Displays the path to the current resource using a hierarchy of links.

## Preview

<ComponentPreview>
  <BreadcrumbExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { Breadcrumb } from "@adn-ui/react";

export default () => (
  <Breadcrumb.Root>
    <Breadcrumb.List>
      <Breadcrumb.Item>
        <Breadcrumb.Link href="/">Home</Breadcrumb.Link>
      </Breadcrumb.Item>
      <Breadcrumb.Separator />
      <Breadcrumb.Item>
        <Breadcrumb.Page>Components</Breadcrumb.Page>
      </Breadcrumb.Item>
    </Breadcrumb.List>
  </Breadcrumb.Root>
);
```

### With Ellipsis
