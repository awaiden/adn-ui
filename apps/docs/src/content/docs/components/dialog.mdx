  <DialogExample.Default client:load />

## Usage

```tsx
import { Dialog, Button } from "@adn-ui/react";

export default () => (
  <Dialog.Root>
    <Dialog.Trigger asChild>
      <Button variant="outline">Edit Profile</Button>
    </Dialog.Trigger>
    <Dialog.Content>
      <Dialog.Header>
        <Dialog.Title>Edit profile</Dialog.Title>
        <Dialog.Description>
          Make changes to your profile here. Click save when you're done.
        </Dialog.Description>
      </Dialog.Header>
      <div className="py-4">Profile content goes here...</div>
      <Dialog.Footer>
        <Button type="submit">Save changes</Button>
      </Dialog.Footer>
    </Dialog.Content>
  </Dialog.Root>
);
```
