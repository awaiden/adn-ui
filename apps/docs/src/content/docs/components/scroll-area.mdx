Augments native scroll functionality for custom, cross-browser styling.

## Preview

<ComponentPreview>
  <ScrollAreaExample.Default client:load />
</ComponentPreview>

## Usage

```tsx
import { ScrollArea } from "@adn-ui/react";

export default () => (
  <ScrollArea.Root className="h-72 w-48 rounded-md border">
    <ScrollArea.Viewport>
      <div className="p-4">Your scrollable content here</div>
    </ScrollArea.Viewport>
    <ScrollArea.Scrollbar orientation="vertical">
      <ScrollArea.Thumb />
    </ScrollArea.Scrollbar>
    <ScrollArea.Corner />
  </ScrollArea.Root>
);
```
